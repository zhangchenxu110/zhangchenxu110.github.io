安装(使用APT安装)

官方安装指南

  1. 导入public key
sudo apt-key adv --keyserver hkp://keyserver.ubuntu.com:80 --recv 0C49F3730359A14518585931BC711F9BA15703C6
  1. 导入源
echo "deb [ arch=amd64,arm64 ] http://repo.mongodb.org/apt/ubuntu xenial/mongodb-org/3.4 multiverse" | sudo tee /etc/apt/sources.list.d/mongodb-org-3.4.list
  1. 更新本地源list
sudo apt update
  1. 安装
sudo apt-get install -y mongodb-org

配置

配置Mongodb /etc/mongod.conf

配置启动文件 /lib/systemd/system/mongod.service

启动

# 启动
sudo service mongod start
# 结束
sudo service mongod stop

如果出现无法启动，先切到路径 /lib/systemd/system
执行：

# 启用 mongodb.service
sudo systemctl enable mongod.service

再

# 启动
sudo service mongod start

卸载

  1. Stop MongoDB.
sudo service mongod stop
  1. Remove Packages.
sudo apt-get purge mongodb-org*
  1. Remove Data Directories.¶
    Remove MongoDB databases and log files.
sudo rm -r /var/log/mongodb
sudo rm -r /var/lib/mongodb

离线安装（解压tgz包的方式）

解压

tar -zxvf mongodb-linux-x86_64-ubuntu1604-3.4.4.tgz

创建存放数据文件和日志的文件夹

mkdir -p /data/db/journal  # 创建数据文件夹
cd /data/db/journal
touch mongodb.log  #创建文件

在安装目录配置文件 mongodb.conf

# cd 到解压路径下
# 创建配置文件
vi mongodb.conf

写入

# 详细记录输出
verbose = true

# 指定服务端口号，默认端口27017
port = 27017

# 指定MongoDB日志文件，注意是指定文件不是目录
logpath = /home/cisdi/DBTest/mongodb/data/db/journal/mongodb.log

# 使用追加的方式写日志
logappend = true

# 指定数据库路径
dbpath = /home/cisdi/DBTest/mongodb/data/db/

#设置每个数据库将被保存在一个单独的目录
directoryperdb = true

# 启用验证
auth = false

#以守护进程的方式运行MongoDB，创建服务器进程
fork = true
#安静输出
quiet = true

配置PATH

sudo vi /etc/profile

追加

export PATH=/home/cisdi/DBTest/mongodb/mongodb-3.4.4/bin:$PATH

执行一下

source /etc/profile

在解压路径下启动mongodb

mongod -f mongodb.conf

关闭mongodb

# 进入mongoshell
mongo

# 使用admin数据库
use admin

# 关闭mongodb服务
db.shutdownServer()
